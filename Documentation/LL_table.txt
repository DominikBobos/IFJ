<prog> -> KEYWORD_DEF TYPE_INENTIFIER(<params>): TYPE_EOL 
		<statement> TYPE_EOL
	  <prog>
<prog> -> TYPE_EOL <prog>
<prog> -> <statement> TYPE_EOL <prog>

<statement> -> KEYWORD_IF <expression>: TYPE_EOL
		<statement>
		KEYWORD_ELSE: TYPE_EOL
		<statement>

<statement> -> KEYWORD_WHILE: <expression>: TYPE_EOL
		<statement>

<statement> -> KEYWORD_RETURN <expression>
<statement> -> TYPE_IDENTIFIER = <expression>
<statement> -> <statement> TYPE_EOL <statement>			//aby mohlo byt viacero statementov zasebou
<statement> -> KEYWORD_PRINT(<expression>)
<statement> -> KEYWORD_PASS

<params> -> TYPE_IDENTIFIER <param_next>
<params> -> ε
<param_next> -> , TYPE_IDENTIFIER <param_next>
<param_next> -> ε

<expression> -> TYPE_IDENTIFIER
<expression> -> 'TYPE_CHR'
<expression> -> 'TYPE_CHR', TYPE_IDENTIFIER , <expression_n>
<expression> -> ε
<expression_n> -> ,<expression_n>
<expression_n> -> ε
