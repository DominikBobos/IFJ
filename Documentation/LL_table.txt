
1. 	<prog> -> KEYWORD_DEF TYPE_IDENTIFIER TYPE_LEFT_PAR<params>TYPE_RIGHT_PAR TYPE_COLON TYPE_EOL TYPE_INDENT <statement> TYPE_EOL TYPE_DEDENT <prog>
2. 	<prog> -> TYPE_EOL <prog>
3. 	<prog> -> <statement> TYPE_EOL <prog>
4. 	<prog> -> TYPE_EOF <end> 							//po kazdom statemente alebo progu sa znova zacina prog, 
											cize posledny krok pri parsovani sa podla mna bude zacinat 
											iba <prog> , dalej ziska token TYPE_EOF a skoci to 
											koncoveho stavu <end> kde uz neparsuje 	

5. 	<statement> -> KEYWORD_IF <expression> TYPE_COLON TYPE_EOL TYPE_INDENT <statement> KEYWORD_ELSETYPE_COLON TYPE_EOL <statement> TYPE_DEDENT
6. 	<statement> -> KEYWORD_WHILE TYPE_COLON <expression> TYPE_COLON TYPE_EOL TYPE_INDENT <statement> TYPE_DEDENT
7. 	<statement> -> KEYWORD_RETURN <expression>
8. 	<statement> -> TYPE_IDENTIFIER TYPE_ASSIGN_VALUE <expression>

9. 	<statement> -> KEYWORD_PRINT TYPE_LEFT_PAR<expression>TYPE_RIGHT_PAR        	//co ked je prazdny print/komentar v printe
10. 	<statement> -> KEYWORD_PASS
11. 	<statement> -> KEYWORD_INPUTS TYPE_LEFT_PAR TYPE_RIGHT_PAR
12. 	<statement> -> KEYWORD_INPUTI TYPE_LEFT_PAR TYPE_RIGHT_PAR
13. 	<statement> -> KEYWORD_INPUTF TYPE_LEFT_PAR TYPE_RIGHT_PAR
14. 	<statement> -> KEYWORD_LEN TYPE_LEFT_PAR TYPE_STRING TYPE_RIGHT_PAR
15. 	<statement> -> KEYWORD_SUBSTR TYPE_LEFT_PAR TYPE_STRING TYPE_COMMA TYPE_INT TYPE_COMMA TYPE_INT TYPE_RIGHT_PAR
16. 	<statement> -> KEYWORD_CHR TYPE_LEFT_PAR TYPE_STRING TYPE_COMMA TYPE_INT TYPE_RIGHT_PAR
17. 	<statement> -> KEYWORD_ORD TYPE_LEFT_PAR TYPE_INT TYPE_RIGHT_PAR

18.	<params> -> TYPE_IDENTIFIER <param_next>
19. 	<params> -> ε
20. 	<param_next> -> TYPE_COMMA TYPE_IDENTIFIER <param_next>
21. 	<param_next> -> ε
22.	<end>-> ε	//riesenie <end> chybalo
